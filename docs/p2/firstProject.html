@Controller
public class BannerController {
    //MVC SpringMVC
    //Controller Model View

    @GetMapping("/test")
    public void test(HttpServletResponse response){
        try {
            response.getWriter().write("Hello World!");
        } catch (IOException e) {
            e.printStackTrace();
        }
//        return "Hello World!";
    }
}

ResponseBody 简化 ServletResponse

@Controller
public class BannerController {
    //MVC SpringMVC
    //Controller Model View

    @GetMapping("/test")
    @ResponseBody
    public String test(){
//            response.getWriter().write("Hello World!");

    //response header text/plain
    //application/json
    //POSTMAN
    //Junit test
        return "Hello World!";
    }
}
